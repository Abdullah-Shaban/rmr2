K-means

##

    dist.fun = 
      function(C, P) {
        apply(C,
              1, 
              function(x) 
                matrix(
                  rowSums((t(t(P) - x))^2), 
                  ncol = length(x)))}

##

    kmeans.map = 
      function(., P) {
        nearest = {
          if(is.null(C)) 
            sample(
              1:num.clusters, 
              nrow(P), 
              replace = T)
          else {
            D = dist.fun(C, P)
            nearest = max.col(-D)}}
        if(!(combine || in.mem.combine))
          keyval(nearest, P) 
        else 
          keyval(nearest, cbind(1, P))}

##

    kmeans.reduce = {
      if (!(combine || in.mem.combine) ) 
        function(x, P) 
          t(as.matrix(apply(P, 2, mean)))
      else 
        function(k, P) 
          keyval(k, t(as.matrix(apply(P, 2, sum))))}

##

kmeans.mr = 
  function(
    P, 
    num.clusters, 
    num.iter, 
    combine, 
    in.mem.combine) {

##

    C = NULL
    for(i in 1:num.iter ) {
      C = 
        values(
          from.dfs(
            mapreduce(
              P, 
              map = kmeans.map,
              reduce = kmeans.reduce)))
      if(combine || in.mem.combine)
        C = C[, -1]/C[, 1]
      points(C, col = i + 1, pch = 19)

##

      if(nrow(C) < num.clusters) {
        C = 
          rbind(
            C,
            matrix(
              rnorm(
                (num.clusters - nrow(C)) * nrow(C)), 
              ncol = nrow(C)) %*% C) }}
        C}

Linear Least Squares

##

\[ \mathbf{X b = y} \] 

  solve(t(X)%*%X, t(X)%*%y)

##

Sum = 
  function(., YY) 
    keyval(1, list(Reduce('+', YY)))

##

XtX = 
  values(
    from.dfs(
      mapreduce(
        input = X,
        map = 
          function(., Xi) 
            keyval(1, list(t(Xi) %*% Xi)),
        reduce = Sum,
        combine = TRUE)))[[1]]

##

Xty = 
  values(
    from.dfs(
      mapreduce(
        input = X,
        map = function(., Xi)
          keyval(1, list(t(Xi) %*% y)),
        reduce = Sum,
        combine = TRUE)))[[1]]
